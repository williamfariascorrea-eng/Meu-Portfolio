SELECT
  u.nome,
  p.descricao AS perfil,
  COUNT(a.id) AS total_acessos
FROM usuarios u
INNER JOIN perfis p ON u.perfil_id = p.id
LEFT JOIN acessos a ON u.id = a.usuario_id
WHERE u.ativo = 1
GROUP BY u.nome, p.descricao
ORDER BY total_acessos DESC;
